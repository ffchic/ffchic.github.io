def tencent(msg):

    APPKEY = '123'
    url = 'https://api.ai.qq.com/fcgi-bin/nlp/nlp_textchat'
    params = {
        'app_id': APPID,

        'nonce_str': ''.join(random.choice(string.ascii_letters + string.digits) for x in range(16)),
