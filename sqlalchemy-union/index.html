Base = declarative_base()

class User(Base):
    __tablename__ = "user"

    id = Column(CHAR(20),primary_key = True)
    name = Column(CHAR(20))
    age = Column(INTEGER)


class Teacher(Base):
    __tablename__ = "teacher"

    id = Column(CHAR(20),primary_key = True)
    tec_name = Column(CHAR(20))
    tec_age = Column(INTEGER)

def insert_data():

    DBSession = sessionmaker(bind=engine)
    session = DBSession()
    for x in range(10):
        temp = {}
        temp['id'] = x
        temp['name'] = 'user_' + str(x)
        temp['age'] = x
        user = User(**temp)
        session.add(user)

    for x in range(15):
        temp = {}
        temp['id'] = x
        temp['tec_name'] = 'tec_' + str(x)
        temp['tec_age'] = x * 2 
        tec = Teacher(**temp)
        session.add(tec)

    session.commit()
    session.close()
    print('success')


if __name__ = '__main__':

    create_table(Teacher)
